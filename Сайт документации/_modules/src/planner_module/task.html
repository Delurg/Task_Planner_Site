class TaskPlanners:
  def __init__(self):
    """
    Инициализация планироващика задач
    """
    self.tasks = []



[документация]
  def CreateTask(self, task: str) -> str:
    """
    Создание задачи

    Args:
      task (str): Строка хранящая имя и даты задачи

    Returns:
      str: Сообщение об успешном создании задачи
    """
    return f"Успешное создание задачи {task}"





[документация]
  def ChangeTask(self, task: str, task_change: str) -> str:
    """
    Изменение задачи

    Args:
      task (str): Строка хранящая старую задачу, которые мы изменяем
      task_change (str): Строка хранящая изменения для задачи

    Returns:
      str: Сообщение об успешном изменении задачи
    """
    self.tasks.remove(task)
    self.tasks.append(task_change)
    return f"Успешное изменение задачи {task_change}"





[документация]
  def NotificationTask(self, task: str) -> str:
    """
    Оповещение о задаче

    Args:
      task (str): Строка хранящая задачу, которую мы передаём для упоминания её польщователю

    Returns:
      str: Сообщение об необходимости выполнить передаваемую задачу
    """
    return f"Внимание необзодимо выполнить задачу {task}!"





[документация]
  def SaveTask(self, task: str) -> str:
    """
    Сохранение задачи

    Args:
      task (str): Строка хранящая задачу, которую мы передаём для сохранения

    Returns:
      str: Сообщение об успешном сохранении задачи
    """
    self.tasks.append(task)
    return f"Успешное сохранение задачи {task}"
